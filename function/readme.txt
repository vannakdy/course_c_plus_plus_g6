

- function ?    
    - none return function 
        - start void type
        - not return value to outside
        - has parameter and no parameter 
        void Display(){
            cout << "Hello My function!!";
        }
        void UserInfo(string name){ // has 1 param
            cout << "Name is :" << name << endl;
        }
        void UserInfo1(string name,int age){ // has 1 param
            cout << "Name is :" << name << endl;
            cout << "Age : " << age << endl;
        }
    - return function
        - start type int , float , double ,string ,.... 
        - type != void 
        - return value to outside
        - has parameter and no parameter 
        int sum(){
            return 100 + 10;
        }
        float getAvg(){
            return 1000/3;
        }
        float getAvg1(double total_score){
            return total_score/3;
        }
